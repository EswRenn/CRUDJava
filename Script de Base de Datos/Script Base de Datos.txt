use EvaluacionPersonal;

create table Empresa(
IdEmpresa numeric (10) identity (1,1),
Nombre varchar(10),
Direccion varchar (10),
Telefono numeric,
Email varchar (10),
constraint PK_IdEmpresa primary key (IdEmpresa)
);
 Select * From Empresa

create table Puesto(
IdPuesto numeric (10) identity (1,1),
Nombre varchar (10),
Caracteristicas varchar(10),
Sueldo numeric,
constraint PK_IdPuesto primary key (IdPuesto)
);
 Select * From Puesto

create table TipoPregunta(
IdTipo numeric (10) identity (1,1),
Tipo1 varchar(10),
Tipo2 varchar (10),
Tipo3 varchar (10),
Tipo4 varchar (10),
constraint PK_IdTipo primary key (IdTipo)
);
 Select * From TipoPregunta

create table PonderacionPregunta(
IdPonderacion numeric (10) identity (1,1),
Punteo1 numeric,
Punteo2 numeric,
Punteo3 numeric,
Punteo4 numeric,
constraint PK_IdPonderacion primary key (IdPonderacion)
);
 Select * From PonderacionPregunta

create table Preguntas(
IdPregunta numeric (10) identity (1,1),
IdTipo numeric(10),
IdPonderacion numeric(10),
Pregunta1 varchar(10),
Pregunta2 varchar(10),
Pregunta3 varchar(10),
Pregunta4 varchar(10),
constraint PK_IdPregunta primary key (IdPregunta),
constraint FK_IdTipo foreign key (IdTipo) references TipoPregunta (IdTipo),
constraint FK_IdPonderacion foreign key (IdPonderacion) references PonderacionPregunta (IdPonderacion)
);
 Select * From Preguntas

create table TipoEvaluacion(
IdEvaluacion numeric (10) identity (1,1),
IdPregunta numeric(10),
Evaluacion1 varchar(10),
Evaluacion2 varchar(10),
constraint PK_IdEvaluacion primary key (IdEvaluacion),
constraint FK_IdPregunta foreign key (IdPregunta) references Preguntas (IdPregunta)
);
 Select * From TipoEvaluacion

create table Empleados(
IdEmpleado numeric (10) identity (1,1),
IdEmpresa numeric(10),
IdPuesto numeric(10),
Nombre varchar(10),
Direccion varchar(10),
Telefono numeric,
Email varchar(10),
constraint PK_IdEmpleado primary key (IdEmpleado),
constraint FK_IdEmpresa foreign key (IdEmpresa) references Empresa (IdEmpresa),
constraint FK_IdPuesto foreign key (IdPuesto) references Puesto (IdPuesto)
);
 Select * From Empleados
 
 create table Usuario (
IdUsuario numeric (10) identity (1,1),
IdEmpleado numeric(10),
Contrase√±a varchar(10),
Email varchar(15),
constraint PK_IdUsuario primary key (IdUsuario),
constraint FK_IdEmpleado foreign key (IdEmpleado) references Empleados (IdEmpleado)
);
 Select * From Usuario

create table RegistroEvaluacionPersonal(
IdRegistro numeric (10) identity (1,1),
IdEmpleado numeric(10),
IdEvaluacion numeric(10),
Respuesta1 varchar(10),
Respuesta2 varchar(10),
Respuesta3 varchar(10),
Respuesta4 varchar(10),
constraint PK_IdRegistro primary key (IdRegistro),
constraint FK_IdEmpleado_E foreign key (IdEmpleado) references Empleados (IdEmpleado),
constraint FK_IdEvaluacion foreign key (IdEvaluacion) references TipoEvaluacion (IdEvaluacion)
);
 Select * From RegistroEvaluacionPersonal

 create table CargaMasiva(
IdCarga numeric (10) identity (1,1),
IdRegistro numeric(10),
IdEmpleado numeric(10),
IdPuesto numeric(10),
constraint PK_IdCarga primary key (IdCarga),
constraint FK_IdRegistro_R foreign key (IdRegistro) references RegistroEvaluacionPersonal (IdRegistro),
constraint FK_IdEmpleado_Em foreign key (IdEmpleado) references Empleados (IdEmpleado),
constraint FK_IdPuesto_P foreign key (Idpuesto) references Puesto (IdPuesto)
);
 Select * From CargaMasiva
